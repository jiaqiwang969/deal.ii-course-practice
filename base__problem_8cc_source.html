151 template <int dim>
152 void
153 BaseProblem<dim>::setup_system()
154 {
155  TimerOutput::Scope timer_section(timer, "setup_system");
156  if (!fe)
157  {
158  fe = FETools::get_fe_by_name<dim>(fe_name);
159  mapping = std::make_unique<MappingQGeneric<dim>>(mapping_degree);
160  const auto vars = dim == 1 ? "x" : dim == 2 ? "x,y" : "x,y,z";
161  forcing_term.initialize(vars, forcing_term_expression, constants);
162  exact_solution.initialize(vars, exact_solution_expression, constants);
163 
164  dirichlet_boundary_condition.initialize(
165  vars, dirichlet_boundary_conditions_expression, constants);
166 
167  neumann_boundary_condition.initialize(
168  vars, neumann_boundary_conditions_expression, constants);
169  }
170 
171  dof_handler.distribute_dofs(*fe);
172 
173  locally_owned_dofs = dof_handler.locally_owned_dofs();
174  DoFTools::extract_locally_relevant_dofs(dof_handler, locally_relevant_dofs);
175 
176  pcout << "Number of degrees of freedom: " << dof_handler.n_dofs()
177  << std::endl;
178 
179 
180  constraints.clear();
181  constraints.reinit(locally_relevant_dofs);
182  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
183 
184  for (const auto &id : dirichlet_ids)
185  VectorTools::interpolate_boundary_values(
186  *mapping, dof_handler, id, dirichlet_boundary_condition, constraints);
187  constraints.close();
188 
189 
190  DynamicSparsityPattern dsp(dof_handler.n_dofs());
191  DoFTools::make_sparsity_pattern(dof_handler, dsp, constraints, false);
192  SparsityTools::distribute_sparsity_pattern(dsp,
193  locally_owned_dofs,
194  mpi_communicator,
195  locally_relevant_dofs);
196 
197 
198  system_matrix.reinit(locally_owned_dofs,
199  locally_owned_dofs,
200  dsp,
201  mpi_communicator);
202 
203  solution.reinit(locally_owned_dofs, mpi_communicator);
204  system_rhs.reinit(locally_owned_dofs, mpi_communicator);
205 
206  locally_relevant_solution.reinit(locally_owned_dofs,
207  locally_relevant_dofs,
208  mpi_communicator);
209 
210  error_per_cell.reinit(triangulation.n_active_cells());
211 
212  // Now call anything that may be needed hook
213  // 可以在此基础上添加扩展，而尽量不改变原基类
214  setup_system_call_back();
215 }

312 template <int dim>
313 void
314 BaseProblem<dim>::estimate()
315 {
316  TimerOutput::Scope timer_section(timer, "estimate");
317  if (estimator_type == "exact")
318  {
319  error_per_cell = 0;
320  QGauss<dim> quad(fe->degree + 1);
321  VectorTools::integrate_difference(*mapping,
322  dof_handler,
323  locally_relevant_solution,
324  exact_solution,
325  error_per_cell,
326  quad,
327  VectorTools::H1_seminorm);
328  }
329  else if (estimator_type == "kelly")
330  {
331  std::map<types::boundary_id, const Function<dim> *> neumann;
332  for (const auto id : neumann_ids)
333  neumann[id] = &neumann_boundary_condition;
334 
335  QGauss<dim - 1> face_quad(fe->degree + 1);
336  KellyErrorEstimator<dim>::estimate(*mapping,
337  dof_handler,
338  face_quad,
339  neumann,
340  locally_relevant_solution,
341  error_per_cell);
342  }
343  else if (estimator_type == "residual")
344  {
345  // h_T || f+\Delta u_h ||_0,T
346  // + \sum over faces
347  // 1/2 (h_F)^{1/2} || [n.\nabla u_h] ||_0,F
348 
349  QGauss<dim - 1> face_quad(fe->degree + 1);
350  QGauss<dim> quad(fe->degree + 1);
351 
352  std::map<types::boundary_id, const Function<dim> *> neumann;
353  for (const auto id : neumann_ids)
354  neumann[id] = &neumann_boundary_condition;
355 
356  KellyErrorEstimator<dim>::estimate(*mapping,
357  dof_handler,
358  face_quad,
359  neumann,
360  locally_relevant_solution,
361  error_per_cell);
362 
363  FEValues<dim> fe_values(*mapping,
364  *fe,
365  quad,
366  update_hessians | update_JxW_values |
367  update_quadrature_points);
368 
369  std::vector<double> local_laplacians(quad.size());
370 
371 
372  double residual_L2_norm = 0;
373 
374  unsigned int cell_index = 0;
375  for (const auto &cell : dof_handler.active_cell_iterators())
376  {
377  fe_values.reinit(cell);
378 
379  fe_values.get_function_laplacians(locally_relevant_solution,
380  local_laplacians);
381  residual_L2_norm = 0;
382  for (const auto q_index : fe_values.quadrature_point_indices())
383  {
384  const auto arg =
385  (local_laplacians[q_index] +
386  forcing_term.value(fe_values.quadrature_point(q_index)));
387  residual_L2_norm += arg * arg * fe_values.JxW(q_index);
388  }
389  error_per_cell[cell_index] +=
390  cell->diameter() * std::sqrt(residual_L2_norm);
391 
392  ++cell_index;
393  }
394  }
395  else
396  {
397  AssertThrow(false, ExcNotImplemented());
398  }
399  auto global_estimator = error_per_cell.l2_norm();
400  error_table.add_extra_column("estimator", [global_estimator]() {
401  return global_estimator;
402  });
403  error_table.error_from_exact(*mapping,
404  dof_handler,
405  locally_relevant_solution,
406  exact_solution);
407 }

413 BaseProblem<dim>::mark()
414 {
415  TimerOutput::Scope timer_section(timer, "mark");
416  if (marking_strategy == "global")
417  {
418  for (const auto &cell : triangulation.active_cell_iterators())
419  cell->set_refine_flag();
420  }
421  else if (marking_strategy == "fixed_fraction")
422  {
423  parallel::distributed::GridRefinement::refine_and_coarsen_fixed_fraction(
424  triangulation,
425  error_per_cell,
426  coarsening_and_refinement_factors.second,
427  coarsening_and_refinement_factors.first);
428  }
429  else if (marking_strategy == "fixed_number")
430  {
431  parallel::distributed::GridRefinement::refine_and_coarsen_fixed_number(
432  triangulation,
433  error_per_cell,
434  coarsening_and_refinement_factors.second,
435  coarsening_and_refinement_factors.first);
436  }
437  else
438  {
439  Assert(false, ExcInternalError());
440  }
441 }

509 template <int dim>
510 void
511 BaseProblem<dim>::run()
512 {
513  print_system_info();
514  make_grid();
515  for (unsigned int cycle = 0; cycle < n_refinement_cycles; ++cycle)
516  {
517  setup_system();
518  assemble_system();
519  solve();
520  estimate();
521  output_results(cycle);
522  if (cycle < n_refinement_cycles - 1)
523  {
524  mark();
525  refine_grid();
526  }
527  }
528  if (pcout.is_active())
529  error_table.output_table(std::cout);
530 }
