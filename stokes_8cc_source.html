66 template <int dim>
67 void
68 Stokes<dim>::assemble_system_one_cell(
69  const typename DoFHandler<dim>::active_cell_iterator &cell,
70  ScratchData & scratch,
71  CopyData & copy)
72 {
73  auto &cell_matrix = copy.matrices[0];
74  auto &cell_rhs = copy.vectors[0];
75 
76  cell->get_dof_indices(copy.local_dof_indices[0]);
77 
78  const auto &fe_values = scratch.reinit(cell);
79  cell_matrix = 0;
80  cell_rhs = 0;
81 
82  for (const unsigned int q_index : fe_values.quadrature_point_indices())
83  {
84  for (const unsigned int i : fe_values.dof_indices())
85  {
86  const auto eps_v = fe_values[velocity].symmetric_gradient(
87  i, q_index); // SymmetricTensor<2,dim>
88  const auto div_v =
89  fe_values[velocity].divergence(i, q_index); // double
90  const auto q = fe_values[pressure].value(i, q_index); // double
91 
92  for (const unsigned int j : fe_values.dof_indices())
93  {
94  const auto eps_u = fe_values[velocity].symmetric_gradient(
95  j, q_index); // SymmetricTensor<2,dim>
96  const auto div_u =
97  fe_values[velocity].divergence(j, q_index); // double
98  const auto p = fe_values[pressure].value(j, q_index); // double
99 
100  cell_matrix(i, j) +=
101  (scalar_product(eps_v, eps_u) - p * div_v - q * div_u) *
102  fe_values.JxW(q_index); // dx
103  }
104  for (const unsigned int i : fe_values.dof_indices())
105  {
106  const auto comp_i = this->fe->system_to_component_index(i).first;
107  cell_rhs(i) +=
108  (fe_values.shape_value(i, q_index) * // phi_i(x_q)
109  this->forcing_term.value(fe_values.quadrature_point(q_index),
110  comp_i) * // f(x_q)
111  fe_values.JxW(q_index)); // dx
112  }
113  }
114  }
115 
116  if (cell->at_boundary())
117  // for(const auto face: cell->face_indices())
118  for (unsigned int f = 0; f < GeometryInfo<dim>::faces_per_cell; ++f)
119  if (this->neumann_ids.find(cell->face(f)->boundary_id()) !=
120  this->neumann_ids.end())
121  {
122  auto &fe_face_values = scratch.reinit(cell, f);
123  for (const unsigned int q_index :
124  fe_face_values.quadrature_point_indices())
125  for (const unsigned int i : fe_face_values.dof_indices())
126  {
127  const auto comp_i =
128  this->fe->system_to_component_index(i).first;
129  cell_rhs(i) +=
130  fe_face_values.shape_value(i, q_index) *
131  this->neumann_boundary_condition.value(
132  fe_face_values.quadrature_point(q_index), comp_i) *
133  fe_face_values.JxW(q_index);
134  }
135  }
136 }

160 template <int dim>
161 void
162 Stokes<dim>::estimate()
163 {
164  TimerOutput::Scope timer_section(this->timer, "estimate");
165  if (this->estimator_type == "kelly")
166  {
167  std::map<types::boundary_id, const Function<dim> *> neumann;
168  for (const auto id : this->neumann_ids)
169  neumann[id] = &this->neumann_boundary_condition;
170 
171  QGauss<dim - 1> face_quad(this->fe->degree + 1);
172  KellyErrorEstimator<dim>::estimate(*this->mapping,
173  this->dof_handler,
174  face_quad,
175  neumann,
176  this->locally_relevant_block_solution,
177  this->error_per_cell,
178  this->fe->component_mask(velocity));
179  }
180  else
181  {
182  AssertThrow(false, ExcNotImplemented());
183  }
184  auto global_estimator = this->error_per_cell.l2_norm();
185  this->error_table.add_extra_column("estimator", [global_estimator]() {
186  return global_estimator;
187  });
188  this->error_table.error_from_exact(*this->mapping,
189  this->dof_handler,
190  this->locally_relevant_block_solution,
191  this->exact_solution);
192 }
