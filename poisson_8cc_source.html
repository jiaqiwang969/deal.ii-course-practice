45 void
46 Poisson<dim>::assemble_system_one_cell(
47  const typename DoFHandler<dim>::active_cell_iterator &cell,
48  ScratchData & scratch,
49  CopyData & copy)
50 {
51  auto &cell_matrix = copy.matrices[0];
52  auto &cell_rhs = copy.vectors[0];
53 
54  cell->get_dof_indices(copy.local_dof_indices[0]);
55 
56  const auto &fe_values = scratch.reinit(cell);
57  cell_matrix = 0;
58  cell_rhs = 0;
59 
60  for (const unsigned int q_index : fe_values.quadrature_point_indices())
61  {
62  for (const unsigned int i : fe_values.dof_indices())
63  for (const unsigned int j : fe_values.dof_indices())
64  cell_matrix(i, j) +=
65  (coefficient.value(fe_values.quadrature_point(q_index)) * // a(x_q)
66  fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
67  fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
68  fe_values.JxW(q_index)); // dx
69  for (const unsigned int i : fe_values.dof_indices())
70  cell_rhs(i) += (fe_values.shape_value(i, q_index) * // phi_i(x_q)
71  this->forcing_term.value(
72  fe_values.quadrature_point(q_index)) * // f(x_q)
73  fe_values.JxW(q_index)); // dx
74  }
75 
76  if (cell->at_boundary())
77  // for(const auto face: cell->face_indices())
78  for (unsigned int f = 0; f < GeometryInfo<dim>::faces_per_cell; ++f)
79  if (this->neumann_ids.find(cell->face(f)->boundary_id()) !=
80  this->neumann_ids.end())
81  {
82  auto &fe_face_values = scratch.reinit(cell, f);
83  for (const unsigned int q_index :
84  fe_face_values.quadrature_point_indices())
85  for (const unsigned int i : fe_face_values.dof_indices())
86  cell_rhs(i) += fe_face_values.shape_value(i, q_index) *
87  this->neumann_boundary_condition.value(
88  fe_face_values.quadrature_point(q_index)) *
89  fe_face_values.JxW(q_index);
90  }
91 }
