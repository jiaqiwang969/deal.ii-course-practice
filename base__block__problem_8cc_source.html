36 template <int dim>
37 void
38 BaseBlockProblem<dim>::setup_system()
39 {
40  TimerOutput::Scope timer_section(this->timer, "setup_system");
41  if (!this->fe) // this 其作用就是指向成员函数所作用的对象
42  {
43  this->fe = FETools::get_fe_by_name<dim>(this->fe_name);
44  this->mapping =
45  std::make_unique<MappingQGeneric<dim>>(this->mapping_degree);
46  const auto vars = dim == 1 ? "x" : dim == 2 ? "x,y" : "x,y,z";
47  this->forcing_term.initialize(vars,
48  this->forcing_term_expression,
49  this->constants);
50  this->exact_solution.initialize(vars,
51  this->exact_solution_expression,
52  this->constants);
53 
54  this->dirichlet_boundary_condition.initialize(
55  vars, this->dirichlet_boundary_conditions_expression, this->constants);
56 
57  this->neumann_boundary_condition.initialize(
58  vars, this->neumann_boundary_conditions_expression, this->constants);
59  }
60 
61  this->dof_handler.distribute_dofs(*this->fe);
62 
63  // 以顺时针的方式重新编号Dofs。
64  std::vector<unsigned int> blocks(this->n_components);
65  unsigned int i = 0;
66  Assert(component_names.size() > 0, ExcInternalError());
67  AssertDimension(this->n_components, component_names.size());
68  blocks[0] = i;
69  for (unsigned int j = 1; j < this->n_components; ++j)
70  {
71  if (component_names[j] == component_names[j - 1])
72  blocks[j] = i;
73  else
74  blocks[j] = ++i;
75  }
76  DoFRenumbering::component_wise(this->dof_handler, blocks);
77 
78  dofs_per_block = DoFTools::count_dofs_per_fe_block(this->dof_handler, blocks);
79 
80  locally_owned_dofs =
81  this->dof_handler.locally_owned_dofs().split_by_block(dofs_per_block);
82 
83 
84  IndexSet non_blocked_locally_relevant_dofs;
85  DoFTools::extract_locally_relevant_dofs(this->dof_handler,
86  non_blocked_locally_relevant_dofs);
87  locally_relevant_dofs =
88  non_blocked_locally_relevant_dofs.split_by_block(dofs_per_block);
89 
90  this->pcout << "Number of degrees of freedom: " << this->dof_handler.n_dofs()
91  << " (" << Patterns::Tools::to_string(dofs_per_block) << ")"
92  << std::endl;
93 
94 
95  this->constraints.clear();
96  this->constraints.reinit(non_blocked_locally_relevant_dofs);
97  DoFTools::make_hanging_node_constraints(this->dof_handler, this->constraints);
98 
99  for (const auto &id : this->dirichlet_ids)
100  VectorTools::interpolate_boundary_values(*this->mapping,
101  this->dof_handler,
102  id,
103  this->dirichlet_boundary_condition,
104  this->constraints);
105  this->constraints.close();
106 
107 
108  TrilinosWrappers::BlockSparsityPattern dsp(locally_owned_dofs,
109  locally_owned_dofs,
110  locally_relevant_dofs,
111  this->mpi_communicator);
112 
113  DoFTools::make_sparsity_pattern(this->dof_handler,
114  dsp,
115  this->constraints,
116  false);
117  // SparsityTools::distribute_sparsity_pattern(dsp,
118  // locally_owned_dofs,
119  // mpi_communicator,
120  // locally_relevant_dofs);
121 
122  dsp.compress();
123  system_block_matrix.reinit(dsp);
124 
125  block_solution.reinit(locally_owned_dofs, this->mpi_communicator);
126  system_block_rhs.reinit(locally_owned_dofs, this->mpi_communicator);
127 
128  locally_relevant_block_solution.reinit(locally_owned_dofs,
129  locally_relevant_dofs,
130  this->mpi_communicator);
131 
132  this->error_per_cell.reinit(this->triangulation.n_active_cells());
133 
134  // 现在调用任何可能需要的东西
135  this->setup_system_call_back();
136 }

165 template <int dim>
166 void
167 BaseBlockProblem<dim>::assemble_system()
168 {
169  TimerOutput::Scope timer_section(this->timer, "assemble_system");
170  QGauss<dim> quadrature_formula(this->fe->degree + 1);
171  QGauss<dim - 1> face_quadrature_formula(this->fe->degree + 1);
172 
173  ScratchData scratch(*this->mapping,
174  *this->fe,
175  quadrature_formula,
176  update_values | update_gradients |
177  update_quadrature_points | update_JxW_values,
178  face_quadrature_formula,
179  update_values | update_quadrature_points |
180  update_JxW_values);
181 
182  CopyData copy(this->fe->n_dofs_per_cell());
183 
184  auto worker = [&](const auto &cell, auto &scratch, auto &copy) {
185  assemble_system_one_cell(cell, scratch, copy);
186  };
187 
188  auto copier = [&](const auto &copy) { copy_one_cell(copy); };
189 
190  using CellFilter =
191  FilteredIterator<typename DoFHandler<dim>::active_cell_iterator>;
192 
193  WorkStream::run(CellFilter(IteratorFilters::LocallyOwnedCell(),
194  this->dof_handler.begin_active()),
195  CellFilter(IteratorFilters::LocallyOwnedCell(),
196  this->dof_handler.end()),
197  worker,
198  copier,
199  scratch,
200  copy);
201 
202  system_block_matrix.compress(VectorOperation::add);
203  system_block_rhs.compress(VectorOperation::add);
204 }
