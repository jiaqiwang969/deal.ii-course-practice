494  par_name = argv[1];
495 
496  if (Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0)
497  deallog.depth_console(2);
498  else
499  deallog.depth_console(0);
500 
501  ProblemType base_problem;
502  base_problem.initialize(par_name);
503  base_problem.run();
504  }
